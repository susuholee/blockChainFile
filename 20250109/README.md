### 연산자

```js
// 단항 산술 연산자 -------------------------------------------------------------------
// +, - , *, /, %
let num = 1;
let num2 = 2;
let num3 = num + num2;
console.log(num3);
// 문자열과 숫자를 더하는경우
// "1" + 1  => 문자열 "11"
// 1 + 1 => 숫자 2 
// 형변환
// 연산자를 공부할때 중요한것은 반환값(쉽게 이해를 돕기위해서 연산자를 사용한 위치에 생성되는 값)과 데이터 타입 (연산자가 반환하는 데이터 타입)

// 비교 연산자 -------------------------------------------------------------------
// 두 개 이상의 값을 비교해서 true와 false의 값을 반환하는 연산자.
let num4 = 3;
let num5 = 4;
// 3 4
num4 == num5; // 두 값이 같니? 3 4 같지 않아 false를 반환 데이터 타입 bool

// 일반 값을 사용해도 되지만 우리가 이후에 호출해서 사용하지 못하는 값이겠죠?.
4 == 4 // 두값이 같니? 4 4 같으니 true를 반환 

// 문자형 데이터와 숫자형 데이터를 비교하는 경우
"4" == 4; // 같다 왜 같냐? 값만 비교를 하기 때문에 자료형은 신경쓰지 않는다. 반환 값이 true
// == 데이터 타입 비교 X 값 비교 O

// 데이터의 타입까지 비교
// === 
// 데이터 타입과 데이터 값까지 둘다 비교
"4" === 4; // false 데이터의 값은 같지만 문자형과 숫자형 데이터 데이터 타입이 다르기때문에

console.log(num4 == num5);

// 부정 연산자
// bool 타입의 값을 반전시키고 싶을때 반대 되는값이 필요한 경우
// 변수의 값을 확인할때
// 사용을 해야하는데 줄일수 있으면 줄이면 가독성이 좀더 좋아진다
// 

console.log(!true); // true의 반대인 false
console.log(!false); // false의 반대인 true

// 비교 연산자와 같이 사용할때
console.log(4 != 4); // 값이 다르니? 달랐을때 true  값이 다른 경우만 비교해서 값이 다르면 true

console.log(4 !== 4); // 값과 타입까지 다른지 비교

// 큰 값과 작은 값을 비교하는 연산자
//  <, > : 값의 크기를 비교하는 연산자. 초과와 미만을 비교할때 3 < 3 3이 3보다 크지 않으니 초과되지 않으니 false     3 < 4  4는 3보다 초과된 값이기 때문에 true
//  <=, >= : 이상과 이하를 비교할때 이상과 이하 3 <= 3 같거나 큰값인지 true  3 <= 4 true 
```

### 조건문
> 개발자의 꽃 조건문
> 걷고 있는데 만보까지만 포인트를 줄거야 앱을 개발한다 하면 
> 걷는 거리가 어느정도 일대마다 포인트를 줄건지 (조건문)


```js
// 조건문 문법
[if 예약어] (조건문) {
    조건이 충족했을대 실행할 스코프(영역)
}

// 실행이 안될때도 있는거냐? 
// 조건문이 충족하지 않았을때 
// true인지 false인지가 조건이다 true
 
// bool true
if(3 < 4) {
    console.log("3은 4보다 커");
}

// 여러개를 추가해도 된다.
// 구현이후에 클린코드

let str = "나는 이수호 학생이야"
if(str === "나는 이준현 학생이야") {
    // 첫번째 조건이 충족했으면 실행하고 종료
    // 조건이 충족하지 않았으면 다음번 조건 확인
}else if(str === "나는 이수호 학생이야") {
    // 조건이 충족되면 실행 하고 종료

// 모든 조건이 충족하지 않을때 기본으로 실행시켜주고싶은 코드가 있어
```

### 반복문
> 반복적으로 실행되어야하는 코드를 작성할때
> 숫자를 1 2 3 4 5 6 증가시켜서 숫자를 출력하고 싶다

```js
let a = 0;
console.log(++a);
console.log(++a);
console.log(++a);
console.log(++a);
console.log(++a);
console.log(++a);
console.log(++a);
console.log(++a);
console.log(++a);
console.log(++a);
// 코드의 간결성, 가독성을 해치는 행위, 확장성

[for 예약어] (선언문; 조건식; 증감식;) {
    // 스코프 영역 코드 작성 영역
}

// i === index 순서를 표현하기 위해서
for(let i = 0; i < 10; i++){
    console.log(i); // 원하는 인덱스만큼 반복되어야할 코드

    // 더이상 실행할 구문 코드 줄이 없으면
}

for(let i = 0; i < 10; i++){
    for(let j = 0; j < 10; j++) {
        console.log(i, j);
    }
}

// let i = 0; 선언문 읽고
// i < 10 조건문 확인
// 코드 영역을 실행하고 
// i++ 증감식 실행
// 조건식 확인
// 코드영역 실행
// 이걸 반복으로 조건문에 충족하지 않을때까지

// 선언문 -> 조건식 -> 코드실행 -> 증감식 -> 조건문 -> 코드실행 -> ( 증감식 -> 조건문 -> 코드실행 )조건문 충족하지 않을때까지 반복
// 1중
// 2중
// 3중 



```
